# CRUD-PHP
CRUD PROJECT USING PHP

<h2>SQL QUERIES THAT USED IN THIS PROJECT:</h2>
CREATE DATABASE employee_management_system;</br>
</br>
CREATE TABLE employees (</br>
    id INT AUTO_INCREMENT PRIMARY KEY,</br>
    name VARCHAR(100) NOT NULL,</br>
    email VARCHAR(100) NOT NULL,</br>
    position VARCHAR(100) NOT NULL,</br>
    department VARCHAR(100) NOT NULL,</br>
    created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,</br>
    updated TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP</br>
);
